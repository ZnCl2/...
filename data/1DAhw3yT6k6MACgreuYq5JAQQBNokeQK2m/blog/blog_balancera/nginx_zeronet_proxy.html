NginX ZeroNet reverse proxy config

location / { proxy_pass http://127.0.0.1:43110; proxy_set_header Host $host; #get rid of media referrer error } location /Websocket { proxy_pass http://127.0.0.1:43110; proxy_http_version 1.1; proxy_read_timeout 1h; #for long live websocket connetion proxy_set_header Upgrade $http_upgrade; proxy_set_header Connection "upgrade"; }

class Window extends JFrame
{
    public Window(String title)
    {
        this.setTitle(title);
        this.setDefaultCloseOperation(EXIT_ON_CLOSE);
        this.setResizable(false);
    }
    
    public void addSurface(Surface srf)
    {
        this.add(srf);
        this.pack();
        this.validate();
    }
}

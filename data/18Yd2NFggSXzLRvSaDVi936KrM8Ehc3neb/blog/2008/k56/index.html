function! s:convert_entity(str)
    let s = a:str
    let s = substitute(s, ‘&’, ‘&’, ‘g’)
    let s = substitute(s, ‘<’, ‘<‘, ‘g’)
    let s = substitute(s, ‘>’, ‘>’, ‘g’)
    set encoding=utf-8
    let s = substitute(s, ‘&#(\d+);‘,’\=nr2char(submatch(1))‘, ‘g’)
    set encoding=cp1251
    let s = iconv( s, ‘utf-8’, ‘cp1251’ )
    return s
endfunction
